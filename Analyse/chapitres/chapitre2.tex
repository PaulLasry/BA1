\chapter{Suites}
\section{Définitions et exemples}
\begin{definition}
  Une suite de nombres réels est une fonction : \begin{align*}
    a :\,\N &\to\R\\ n&\to a(n) = a_n
  \end{align*}
  \textbf{Notation :} $(a_n)_{n\in\N} = (a_n)_{n\ge0} = (a_n)_\N = a_n = (a_0,\, a_1, \ldots,\,a_n)$
\end{definition}
